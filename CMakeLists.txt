
cmake_minimum_required (VERSION 3.8)
set(CXX_STANDARD 20)
set(PROFILING TRUE)
set(USING_VISUAL_STUDIO TRUE)
project ("RayTracer")

include_directories(include)
include_directories(include/Math)
include_directories(include/Object)
include_directories(include/Object/GraphicalObject)
include_directories(include/Raycast)
include_directories(glm)
include_directories(SDL2/include)
link_directories(SDL2/lib/x64)

add_executable (${PROJECT_NAME} "src/main.cpp" "src/Object/GraphicalObject.cpp" "src/Object/Object.cpp" "src/Ray/Ray.cpp"
"src/Ray/Raycast.cpp" "src/Camera.cpp" "src/SDLDisplayer.cpp"  "src/Input.cpp" "src/Math/Vec3.cpp" "src/Benchmark.cpp" "include/Action.h")

if(${USING_VISUAL_STUDIO})
  target_link_libraries(RayTracer SDL2)
  target_link_libraries(RayTracer SDL2main)
else()
  find_package(PkgConfig REQUIRED)
  pkg_check_modules(SDL_PKG REQUIRED IMPORTED_TARGET sdl2)
  target_link_libraries(RayTracer PUBLIC PkgConfig::SDL_PKG)
  set(CMAKE_BUILD_TYPE Release)
  set(CMAKE_CXX_FLAGS "-Wall -Wextra")
  set(CMAKE_CXX_FLAGS_DEBUG "	-g")
  set(CMAKE_CXX_FLAGS_RELEASE "-O3")
endif()
